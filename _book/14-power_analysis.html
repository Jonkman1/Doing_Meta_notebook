14.1 Fixed-Effect Model

# Defineer assumpties
theta <- 0.2
K <- 10
n1 <- 25
n2 <- 25

# Calculeer gepoolde effect standaard fout
sigma <- sqrt(((n1+n2)/(n1*n2)+(theta^2/(2*n1+n2)))/K)

# Calculeer z
z = theta/sigma

# Calculeer de power
1 - pnorm(1.96-z) + pnorm(-1.96-z)
